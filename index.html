អំពី C++

C++ គឺជាភាសាកម្មវិធីមួយដែលសម្បូរបែបនៅក្នុងប្រព័ន្ធកម្មវិធី។ វាជាសិល្បៈនៃការកំណត់និងដំណើរការប្រព័ន្ធមួយដោយប្រើលទ្ធផល និង OOP។

មូលដ្ឋាន C++

កូដដំបូង

យើងចាប់ផ្ដើម C++ ដោយប្រើ #include <iostream> និងមុខងារ main()។

#include <iostream>

int main() {
    std::cout << "សួស្តី ពិភពលោក!" << std::endl;
    return 0;
}

លទ្ធផល:

សួស្តី ពិភពលោក!

ប្រភេទទិន្នន័យ និងអថេរ

ប្រភេទទិន្នន័យមានដូចជា int, float, double, និង string ។ ឧទាហរណ៍៖

#include <iostream>
#include <string>

int main() {
    int age = 25;
    double height = 1.75;
    std::string name = "Chantha";
    std::cout << "ឈ្មោះ: " << name << ", អាយុ: " << age << ", កម្ពស់: " << height << " ម៉ែត្រ" << std::endl;
    return 0;
}

លទ្ធផល:

ឈ្មោះ: Chantha, អាយុ: 25, កម្ពស់: 1.75 ម៉ែត្រ

ការបញ្ចូល និងបង្ហាញទិន្នន័យ

ប្រើ std::cin ដើម្បីបញ្ចូលទិន្នន័យ និង std::cout ដើម្បីបង្ហាញទិន្នន័យ។

#include <iostream>

int main() {
    int age;
    std::cout << "បញ្ចូលអាយុរបស់អ្នក: ";
    std::cin >> age;
    std::cout << "អាយុរបស់អ្នកគឺ " << age << std::endl;
    return 0;
}

លទ្ធផល:

បញ្ចូលអាយុរបស់អ្នក: 30
អាយុរបស់អ្នកគឺ 30

ប្រមាណវិធី

ប្រមាណវិធីសម្រាប់គណនា: +, -, *, /, % ។

#include <iostream>

int main() {
    int a = 10, b = 3;
    std::cout << "ផលបូក: " << a + b << std::endl;
    std::cout << "ផ្លូវចែក: " << a / b << std::endl;
    std::cout << "សំណល់: " << a % b << std::endl;
    return 0;
}

លទ្ធផល:

ផលបូក: 13
ផ្លូវចែក: 3
សំណល់: 1

គ្រប់គ្រងលូប និងលក្ខខណ្ឌ

If Statement

ប្រើ if ដើម្បីត្រួតពិនិត្យលទ្ធផល

#include <iostream>

int main() {
    int score = 90;
    if (score >= 80) {
        std::cout << "ជាប់!" << std::endl;
    } else {
        std::cout << "សូមព្យាយាមម្តងទៀត!" << std::endl;
    }
    return 0;
}

លទ្ធផល:

ជាប់!

Switch

ការប្រើប្រាស់​ switch ដើម្បីគ្រប់គ្រងលទ្ធផលខុសគ្នា នៅពេលមានលក្ខខណ្ឌជាច្រើន។

#include <iostream>
        
        int main() {
            int day = 3;
        
            switch (day) {
                case 1:
                    std::cout << "ថ្ងៃច័ន្ទ" << std::endl;
                    break;
                case 2:
                    std::cout << "ថ្ងៃអង្គារ" << std::endl;
                    break;
                case 3:
                    std::cout << "ថ្ងៃពុធ" << std::endl;
                    break;
                case 4:
                    std::cout << "ថ្ងៃព្រហស្បតិ៍" << std::endl;
                    break;
                case 5:
                    std::cout << "ថ្ងៃសុក្រ" << std::endl;
                    break;
                case 6:
                    std::cout << "ថ្ងៃសៅរ៍" << std::endl;
                    break;
                case 7:
                    std::cout << "ថ្ងៃអាទិត្យ" << std::endl;
                    break;
                default:
                    std::cout << "មិនមែនជាថ្ងៃដែលត្រឹមត្រូវទេ" << std::endl;

            return 0;
        }

អារេ (Arrays)

អារេមួយទិស (Single-Dimensional Array)

អារេមួយទិសអាចផ្ទុកតម្លៃជាច្រើនជាប់លាប់គ្នា។

#include <iostream>

int main() {
    int numbers[5] = {10, 20, 30, 40, 50};
    for (int i = 0; i < 5; i++) {
        std::cout << "Value at index " << i << ": " << numbers[i] << std::endl;
    }
    return 0;
}

លទ្ធផល:

Value at index 0: 10
Value at index 1: 20
Value at index 2: 30
Value at index 3: 40
Value at index 4: 50

អារេពីរទិស (Multi-Dimensional Array)

អារេពីរទិសអាចផ្ទុកតម្លៃដូចជាតារាង។

#include <iostream>

int main() {
    int matrix[2][3] = {
        {1, 2, 3},
        {4, 5, 6}
    };
    for (int i = 0; i < 2; i++) {
        for (int j = 0; j < 3; j++) {
            std::cout << "Value at (" << i << "," << j << "): " << matrix[i][j] << " ";
        }
        std::cout << std::endl;
    }
    return 0;
}

លទ្ធផល:

Value at (0,0): 1 Value at (0,1): 2 Value at (0,2): 3 
Value at (1,0): 4 Value at (1,1): 5 Value at (1,2): 6

អារេមិនថេរ (Dynamic Array) ដោយប្រើ Pointer

អាចបង្កើតអារេមិនថេរដែលអាចផ្លាស់ប្តូរទំហំ។

#include <iostream>

int main() {
    int size;
    std::cout << "បញ្ចូលទំហំអារេ: ";
    std::cin >> size;

    int* arr = new int[size];
    for (int i = 0; i < size; i++) {
        arr[i] = i * 2;
    }

    std::cout << "អារេត្រូវបានបង្កើតជោគជ័យ:" << std::endl;
    for (int i = 0; i < size; i++) {
        std::cout << arr[i] << " ";
    }

    delete[] arr; // លុបអារេដើម្បីទាន់សម័យ
    return 0;
}

លទ្ធផល:

បញ្ចូលទំហំអារេ: 5
អារេត្រូវបានបង្កើតជោគជ័យ:
0 2 4 6 8

Object-Oriented Programming (OOP)

Classes និង Objects

Classes ជាទម្រង់ដែលបញ្ជាក់ពីមុខងារ និងលក្ខណៈរបស់ Object មួយ។

#include <iostream>

class Car {
public:
    std::string brand;
    int year;

    void displayInfo() {
        std::cout << "Brand: " << brand << ", Year: " << year << std::endl;
    }
};

int main() {
    Car car1;
    car1.brand = "Toyota";
    car1.year = 2023;
    car1.displayInfo();

    return 0;
}

លទ្ធផល:

Brand: Toyota, Year: 2023

Constructor និង Destructor

Constructor ត្រូវបានអនុវត្តនៅពេល Object ត្រូវបានបង្កើត និង Destructor នៅពេល Object នោះត្រូវបានបំផ្លាញ។

#include <iostream>

class Car {
public:
    Car() {
        std::cout << "Car Created!" << std::endl;
    }

    ~Car() {
        std::cout << "Car Destroyed!" << std::endl;
    }
};

int main() {
    Car car1;
    return 0;
}

លទ្ធផល:

Car Created!
Car Destroyed!

Inheritance

Inheritance អនុញ្ញាតឱ្យបង្កើត Class ថ្មីដោយផ្អែកលើ Class មួយផ្សេងទៀត។

#include <iostream>

class Vehicle {
public:
    int speed = 0;
    void accelerate(int increment) {
        speed += increment;
        std::cout << "Speed: " << speed << " km/h" << std::endl;
    }
};

class Car : public Vehicle {
public:
    void honk() {
        std::cout << "ប៊ីបប៊ីប!" << std::endl;
    }
};

int main() {
    Car myCar;
    myCar.accelerate(20);
    myCar.honk();
    return 0;
}

លទ្ធផល:

Speed: 20 km/h
ប៊ីបប៊ីប!

Polymorphism

Polymorphism អនុញ្ញាតឱ្យ method នឹងមានឈ្មោះដូចគ្នាតែមានពិការការងារផ្សេងគ្នា។

#include <iostream>

class Animal {
public:
    virtual void sound() {
        std::cout << "Animal sound!" << std::endl;
    }
};

class Dog : public Animal {
public:
    void sound() override {
        std::cout << "Woof!" << std::endl;
    }
};

class Cat : public Animal {
public:
    void sound() override {
        std::cout << "Meow!" << std::endl;
    }
};

int main() {
    Animal* animal1 = new Dog();
    Animal* animal2 = new Cat();

    animal1->sound();
    animal2->sound();

    delete animal1;
    delete animal2;
    return 0;
}

លទ្ធផល:

Woof!
Meow!

ពហុមុខងារនៃ OOP (Advanced OOP Concepts)

Encapsulation

Encapsulation គឺជាការការពារទិន្នន័យនៅក្នុង Class ហើយមានការគ្រប់គ្រងចូលប្រើតាមវិធីសាស្ត្រ។

#include <iostream>

class BankAccount {
private:
    double balance;

public:
    BankAccount(double initialBalance) {
        balance = initialBalance;
    }

    void deposit(double amount) {
        if (amount > 0) {
            balance += amount;
            std::cout << "Deposited: " << amount << ", New Balance: " << balance << std::endl;
        }
    }

    void withdraw(double amount) {
        if (amount <= balance) {
            balance -= amount;
            std::cout << "Withdrawn: " << amount << ", New Balance: " << balance << std::endl;
        } else {
            std::cout << "Insufficient funds!" << std::endl;
        }
    }

    double getBalance() const {
        return balance;
    }
};

int main() {
    BankAccount account(1000.0);
    account.deposit(500);
    account.withdraw(200);
    std::cout << "Final Balance: " << account.getBalance() << std::endl;
    return 0;
}

លទ្ធផល:

Deposited: 500, New Balance: 1500
Withdrawn: 200, New Balance: 1300
Final Balance: 1300

Operator Overloading

Operator Overloading អនុញ្ញាតឱ្យប្រើប្រមាណវិធីសម្រាប់ Class ដែលបានកំណត់ដោយអ្នកប្រើ។

#include <iostream>

class Complex {
private:
    double real, imag;

public:
    Complex(double r = 0, double i = 0) : real(r), imag(i) {}

    Complex operator + (const Complex &other) {
        return Complex(real + other.real, imag + other.imag);
    }

    void display() const {
        std::cout << real << " + " << imag << "i" << std::endl;
    }
};

int main() {
    Complex c1(3.0, 2.0), c2(1.5, 4.5);
    Complex c3 = c1 + c2;
    c3.display();
    return 0;
}

លទ្ធផល:

4.5 + 6.5i

Abstract Classes និង Polymorphism

Abstract Class មានមុខងារដែលមិនបានអនុវត្ត (pure virtual function) ហើយត្រូវបានអនុវត្តក្នុង Class កូន។

#include <iostream>

class Shape {
public:
    virtual void draw() = 0; // Pure virtual function
};

class Circle : public Shape {
public:
    void draw() override {
        std::cout << "Drawing Circle" << std::endl;
    }
};

class Rectangle : public Shape {
public:
    void draw() override {
        std::cout << "Drawing Rectangle" << std::endl;
    }
};

int main() {
    Shape* shape1 = new Circle();
    Shape* shape2 = new Rectangle();

    shape1->draw();
    shape2->draw();

    delete shape1;
    delete shape2;
    return 0;
}

លទ្ធផល:

Drawing Circle
Drawing Rectangle

ប្រធានបទកម្រិតខ្ពស់ (Advanced C++ Topics)

Templates

Templates អនុញ្ញាតឱ្យបង្កើត Class ឬ Function មួយដែលអាចប្រើជាមួយប្រភេទទិន្នន័យផ្សេងៗ។

#include <iostream>

template <typename T>
T add(T a, T b) {
    return a + b;
}

int main() {
    std::cout << "Sum of integers: " << add(3, 4) << std::endl;
    std::cout << "Sum of floats: " << add(3.5, 4.5) << std::endl;
    return 0;
}

លទ្ធផល:

Sum of integers: 7
Sum of floats: 8.0

Exception Handling

Exception Handling គឺជាវិធីសាស្ត្រដើម្បីដោះស្រាយកំហុសក្នុងកម្មវិធីដោយប្រើ try, catch និង throw។

#include <iostream>

double divide(double a, double b) {
    if (b == 0) {
        throw "Division by zero error!";
    }
    return a / b;
}

int main() {
    try {
        std::cout << "10 / 2 = " << divide(10, 2) << std::endl;
        std::cout << "10 / 0 = " << divide(10, 0) << std::endl;
    } catch (const char* msg) {
        std::cout << "Error: " << msg << std::endl;
    }
    return 0;
}

លទ្ធផល:

10 / 2 = 5
Error: Division by zero error!

File Input/Output (File I/O)

File I/O អនុញ្ញាតឱ្យកម្មវិធីអាន និងសរសេរទិន្នន័យទៅក្នុងឯកសារ។

#include <iostream>
#include <fstream>

int main() {
    // Writing to a file
    std::ofstream outFile("example.txt");
    outFile << "Hello, World!" << std::endl;
    outFile << "C++ File I/O Example" << std::endl;
    outFile.close();

    // Reading from a file
    std::ifstream inFile("example.txt");
    std::string line;
    std::cout << "Content of example.txt:" << std::endl;
    while (std::getline(inFile, line)) {
        std::cout << line << std::endl;
    }
    inFile.close();
    return 0;
}

លទ្ធផល:

Content of example.txt:
Hello, World!
C++ File I/O Example
